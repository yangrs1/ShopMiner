## 四、接口自动化测试设计

### 4.1 核心设计理念

**一个YAML实现不同测试用例，方便维护测试框架**

```
新增测试用例 → 只修改YAML文件（90%场景）
修改断言逻辑 → 微调pytest脚本（10%场景）
重构接口     → 更新YAML字段映射
```

### 4.3 request_util.py 设计

```python
class RequestUtil:
    """requests封装 - 统一请求/响应/日志/断言"""

    def __init__(self, base_url, default_headers=None):
        self.base_url = base_url
        self.session = requests.Session()
        self.session.headers.update(default_headers or {})

    def request(self, method, path, **kwargs):
        url = f"{self.base_url}{path}"
        response = self.session.request(method, url, **kwargs)
        self._log_request_response(method, url, kwargs, response)
        return response

    def assert_response(self, response, expected):
        """统一断言：状态码 + 业务码 + 字段包含 + 字段值"""
        assert response.status_code == expected["status_code"]
        data = response.json()
        if "code" in expected:
            assert data["code"] == expected["code"]
        if "message_contains" in expected:
            assert expected["message_contains"] in data.get("message", "")
        if "data_contains" in expected:
            for field in expected["data_contains"]:
                assert field in data.get("data", {})
        if "data_equals" in expected:
            for key, value in expected["data_equals"].items():
                assert data["data"][key] == value
```

### 4.5 通用pytest测试模板

```python
"""{模块名} API测试 - YAML数据驱动"""
import pytest
import allure
from utils.yaml_loader import load_yaml

DATA = load_yaml("tests/data/yaml/{module}.yaml")


@allure.feature("{模块名}")
class Test{Module}:

    @allure.story("{功能}成功")
    @allure.severity(allure.severity_level.CRITICAL)
    @pytest.mark.parametrize("case", DATA["{feature}_success"])
    def test_{feature}_success(self, client, case):
        with allure.step(f"执行: {case['name']}"):
            resp = client.{method}(
                "{path}",
                json=case.get("json"),
                headers=...,  # 根据需要注入
            )
        with allure.step("断言响应"):
            assert resp.status_code == case["expected"]["status_code"]
            allure.attach(resp.text, "响应内容", allure.attachment_type.JSON)
```

```python
class BasePage:
    """基础页面对象 - 封装Playwright通用操作"""

    def __init__(self, page):
        self.page = page

    def navigate(self, path="/"):
        self.page.goto(f"http://localhost:5000{path}")

    def fill(self, selector, value):
        self.page.fill(selector, value)

    def click(self, selector):
        self.page.click(selector)

    def get_text(self, selector):
        return self.page.text_content(selector)

    def is_visible(self, selector):
        return self.page.is_visible(selector)

    def wait_for_selector(self, selector, timeout=5000):
        self.page.wait_for_selector(selector, timeout=timeout)

    def screenshot(self, name="screenshot"):
        self.page.screenshot(path=f"tests/report/screenshots/{name}.png")
```

```python
class DBUtil:
    """数据库验证工具 - 测试后验证数据一致性"""

    def __init__(self, session):
        self.session = session

    def count_rows(self, model_class, **filters):
        query = self.session.query(model_class)
        for key, value in filters.items():
            query = query.filter(getattr(model_class, key) == value)
        return query.count()

    def get_user_by_email(self, email):
        from app.models.user import User
        return self.session.query(User).filter_by(email=email).first()

    def get_order_by_id(self, order_id):
        from app.models.order import Order
        return self.session.query(Order).get(order_id)
```

### 6.3 assertion_util.py 设计

```python
class AssertionUtil:
    """高级断言工具"""

    @staticmethod
    def assert_json_schema(response_json, schema):
        """JSON Schema验证"""
        from jsonschema import validate
        validate(instance=response_json, schema=schema)

    @staticmethod
    def assert_response_time(response, max_seconds=2.0):
        """响应时间断言"""
        assert response.elapsed.total_seconds() < max_seconds

    @staticmethod
    def assert_pagination(response_json, expected_page=1):
        """分页结构断言"""
        data = response_json.get("data", {})
        assert "total" in data
        assert "pages" in data
        assert data["page"] == expected_page

    @staticmethod
    def assert_field_types(data, type_map):
        """字段类型断言"""
        for field, expected_type in type_map.items():
            assert field in data, f"缺少字段: {field}"
            assert isinstance(data[field], expected_type), \
                f"字段{field}类型错误: 期望{expected_type}, 实际{type(data[field])}"
```
